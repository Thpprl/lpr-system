6. ตัวอย่างโค้ดโปรแกรม
================================================================================

  ---- run.py (จุดเริ่มต้นรันระบบ) ----

    import sys
    from pipeline.pipeline import run_pipeline
    from pipeline.batch import run_pipeline_batch
    from pipeline.postprocess.formatter import format_lpr_result
    from pipeline.utils.saver import save_json, save_csv

    if __name__ == "__main__":
        mode = sys.argv[1]  # "single" หรือ "batch"
        path = sys.argv[2]  # path รูปภาพ หรือโฟลเดอร์
        save = "--save" in sys.argv

        if mode == "single":
            result = run_pipeline(path)
            print(format_lpr_result(result))
            if save:
                save_json(result)
                save_csv(result)

        elif mode == "batch":
            results = run_pipeline_batch(path)
            for r in results:
                print(format_lpr_result(r))
                if save:
                    save_json(r)
                    save_csv(r)


  ---- pipeline/yolo/yolo_crops.py (ตรวจจับป้ายด้วย YOLOv8) ----

    from ultralytics import YOLO
    from pathlib import Path
    import cv2

    MODEL_PATH = "models/yolo/best.pt"
    OUTPUT_DIR = Path("output/crops")
    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

    model = YOLO(MODEL_PATH)

    def crop_plate(image_path: str) -> list[str]:
        img = cv2.imread(image_path)
        results = model(img)

        saved_paths = []
        for i, box in enumerate(results[0].boxes.xyxy):
            x1, y1, x2, y2 = map(int, box)
            crop = img[y1:y2, x1:x2]
            out_path = OUTPUT_DIR / f"plate_{i}.jpg"
            cv2.imwrite(str(out_path), crop)
            saved_paths.append(str(out_path))

        return saved_paths
